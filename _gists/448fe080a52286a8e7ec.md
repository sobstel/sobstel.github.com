<strong>convert_csv_to_sql.php</strong>

    <?php
    $file = $argv[1];
    $table = $argv[2];
    
    $ls = file($file);
    
    $out = "";
    
    foreach ($ls as $l) {
      $d = str_getcsv(trim($l));
      $q = sprintf('INSERT INTO `%s` VALUES ("%s");', $table, implode('","', $d)).PHP_EOL;
      echo $q;
      $out .= $q;
    }
    
    file_put_contents($table.'.sql', $out);
