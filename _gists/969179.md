<strong>ternary_operator_gotcha.php</strong>

    <?php
    // old way
    $value = $value ? $value : false;
    
    // new way
    $value = $value ?: false;
    
    // however...
    $value = isset($value) ?: false; // if eg. $value="value", then it returns (bool)true, and not (string)"value"
