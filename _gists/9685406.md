<strong>auth-vhost.conf</strong>

    <VirtualHost *>
      ServerName auth.dev
      DocumentRoot "/var/www/auth"
      ProxyPassMatch ^(/.*\.php)$ fcgi://127.0.0.1:9000/var/www/auth
      SetEnvIf Authorization "(.*)" HTTP_AUTHORIZATION=$1
    </VirtualHost>
