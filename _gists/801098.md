<strong>function_in_method.php</strong>

    <?php
    class FooClass {
    	public function foo_method() {
    		function foo_func() {
    			echo "it works, really";
    		}
    	}
    }
    
    $foo = new FooClass;
    $foo->foo_method().foo_func();
