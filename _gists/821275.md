    # Source: http://sinatra.inf.ug.edu.pl/rails4/konfiguracja
    # File: ~/.irbc
    require 'rubygems'
    require 'wirble'
    require 'hirb'
    Wirble.init
    Wirble.colorize
    Hirb.enable
    if ENV.include?('RAILS_ENV') && !Object.const_defined?('RAILS_DEFAULT_LOGGER')
      require 'logger'
      RAILS_DEFAULT_LOGGER = Logger.new(STDOUT)
    end
    
    # For rails3 apps
    # File: Gemfile
    group :development do
      gem 'hirb'
      gem 'wirble'
    end
