<strong>preg_replace_callback_match_counter.php</strong>

    <?php
    $counter = 0;
    
    $result = preg_replace_callback(
      '#\?#',
      function ($match) use (&$counter) {
        $counter += 1;
        return $counter;
      },
      'a: ?, b: ?, c: ?, d: ?. e: ?'
    );
    
    echo $result;
    
    // => a: 1, b: 2, c: 3, d: 4. e: 5
